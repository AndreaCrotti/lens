(<$!>) :: Monad m => (a -> b) -> m a -> m bSource

A strict version of (<$>) for monads. 

>>> (+1) <$!> [1,2,3,4]
[2,3,4,5]

(<$!) :: Monad m => b -> m a -> m bSource

A strict version of (<$) for monads. 

>>> () <$! [1,2,3,4]
[(),(),(),()]
